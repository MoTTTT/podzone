    NAME varchar(100),
    EMAIL varchar(100),
    SUBJECT varchar(100),
    MESSAGE varchar(400),
    COMPANY varchar(100),
    ADDRESS varchar(100),
    CONTACT varchar(100),
    PRIMARY KEY (ID));
